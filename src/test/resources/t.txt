PROCEDURE change_code_to_PHONE IS

BEGIN
  
UPDATE UNIT SET CODE='PHONE-001' WHERE CODE='0001';
COMMIT;
UPDATE STORE_POOL SET CODE='PHONE-001' WHERE CODE='0001';
COMMIT;
UPDATE SHIPMENT SET CODE='PHONE-001' WHERE CODE='0001';
COMMIT;
UPDATE ST_DOCS_INV SET R_INVCODE='PHONE-001' WHERE R_INVCODE='0001';
COMMIT;
UPDATE INVCODES SET INVCODE='PHONE-001', REAL_CODE='PHONE-001' WHERE INVCODE='0001';
COMMIT;
UPDATE INVCODES SET INVCODE='TABLET-001'  WHERE REAL_CODE='TABLET-001';
COMMIT;
UPDATE DIC_DATA SET TERM='Sale of SIM-card / phone Телефонная SIM PHONE-001' WHERE UP=9 AND LANG=1 AND CODE=553;
COMMIT;
UPDATE DIC_DATA SET TERM='Продажа Sim карта / р.телефон Телефонная SIM PHONE-001' WHERE UP=9 AND LANG=2 AND CODE=553;
COMMIT;
UPDATE DIC_DATA SET TERM='Rental of SIM-card/ phone Телефонная SIM PHONE-001' WHERE UP=9 AND LANG=1 AND CODE=555;
COMMIT;
UPDATE DIC_DATA SET TERM='Аренда Sim карта / р.телефон Телефонная SIM PHONE-001' WHERE UP=9 AND LANG=2 AND CODE=555;
COMMIT;

UPDATE DIC_DATA SET TERM='Телефонная SIM' WHERE UP=188 AND CODE=1;
COMMIT;

END;
